#   Form

คุณสามารถควบคุม Style ของ Form Element เช่น <input>, <select>, <textarea> ได้โดยง่าย ด้วย Class .bio-input, .bio-select, .bio-textarea ที่ Container

Markup

<!-- input Markup -->
<div class="bio-input">
  <input type="text" placeholder="Type Something">
</div>

<!-- textarea Markup -->
<div class="bio-textarea">
  <textarea rows="3" placeholder="Textarea"></textarea>
</div>

<!-- select Markup -->
<div class="bio-select">
  <select>
    <option disabled>Select dropdown</option>
    <option>Option 1</option>
    <option>Option 2</option>
    <option>Option 3</option>
  </select>
</div>

Field

คุณสามารถใส่ Class .bio-field ที่ Container ด้านนอก เพื่อช่วยจัดการควบคุมเรื่องของการแสดงผลได้ และใส่หัวข้อ กับคำอธิบายเพิ่มเติมได้

การกำหนดสีทำเพื่อใช้แสดงความรู้สึกเพิ่มเติมของข้อมูล เช่น ข้อมูลถูกต้อง หรือ ข้อมูลผิดพลาด เป็นต้น คุณสามารถประยุกต์ใช้สี Semantic Color ได้ตามที่คุณต้องการ

อีกทั้งคุณยังสามารถใส่ <label> เพื่อแสดงหัวข้อของ Field และใส่คำอธิบายเพิ่มเติมได้ (เราแนะนำให้ใช้ <small>)

ตัวอย่างการใช้ Field

<!-- Positive text input -->
<div class="bio-field -positive">
  <label><strong>Name</strong></label>
  <div class="bio-input -positive">
    <input type="text" placeholder="What's your name again?" value="BioMaRu">
  </div>
  <small>Great, awesome name!</small>
</div>

<!-- Warning number input -->
<div class="bio-field -warning">
  <label><strong>Age</strong></label>
  <div class="bio-input">
    <input type="Number" placeholder="Tell me your age">
  </div>
  <small>Shhhhhh!</small>
</div>

<!-- Accent password input -->
<div class="bio-field -accent">
  <label><strong>Password</strong></label>
  <div class="bio-input -positive">
    <input type="password" placeholder="It's secret">
  </div>
</div>

<!-- Disabled text input -->
<div class="bio-field -disabled">
  <label><strong>Under Construction</strong></label>
  <div class="bio-input">
    <input type="Number" placeholder="Tell me your age" disabled>
  </div>
  <small>This field is not ready yet</small>
</div>

<!-- Negative textarea -->
<div class="bio-field -negative">
  <label><strong>Description</strong></label>
  <div class="bio-textarea">
    <textarea rows="3" placeholder="Tell us about you"></textarea>
  </div>
  <small>Sorry, it blank!! Please fill some data</small>
</div>

<!-- Info select -->
<div class="bio-field -info">
  <label><strong>Optional</strong></label>
  <div class="bio-select">
    <select>
      <option disabled>Select dropdown</option>
      <option>Option 1</option>
      <option>Option 2</option>
      <option>Option 3</option>
    </select>
  </div>
  <small><i class="material-icons _fs-6">info_outline</i> The secret is... You can leave this field as a blank</small>
</div>
